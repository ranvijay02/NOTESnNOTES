create procedure myproc(
IN empid int,
OUT allow int
)
begin
 declare dpt int(5);
 declare ln int(5);
   select e.dept_id into dpt from employee e where e.id=empid;
   select d.location_id into ln from dept d where d.id=dpt;
   select l.allowance into allow from location l where l.id=ln;
end;
$$
