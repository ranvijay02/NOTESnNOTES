---------------STORED PROCEDURE ---------
create or replace procedure myproc(
empid IN number,
allw OUT number
)
as
 dpt number;
 ln number;
begin
   select e.dept into dpt from employee e where e.id=empid;
   select d.location into ln from dept d where d.id=dpt;
   select l.allowance into allw from location l where l.id=ln;
end;
